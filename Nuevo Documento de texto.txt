template>
    <!DOCTYPE html>
    <html lang="es">
        <head>
            <title>Cargos</title>
        </head>

        <body>
            <div class="container col-md-7 align-center mb-3">
                <div class="card card border-success">
                    <div class="card-header">Crear cargos</div>
                    <div class="card-body">
                        <form @submit.prevent="load">
                            <div class="mb-3">
                                <label for="cargoname" class="form-label"
                                    >Nombre</label
                                >
                                <input
                                    v-model="cargo"
                                    type="text"
                                    class="form-control"
                                    name="cargoname"
                                    id="cargoname"
                                    aria-describedby="helpId"
                                    placeholder="Cargo"
                                />
                            </div>

                            <div class="mb-3">
                                <label for="descripcion" class="form-label"
                                    >Descripcion</label
                                >
                                <input
                                    v-model="desc"
                                    type="text"
                                    class="form-control"
                                    name="descripcion"
                                    id="descripcion"
                                    aria-describedby="helpId"
                                    placeholder="Descripcion"
                                />
                            </div>
                            <div
                                class="d-grid gap-2 d-md-flex justify-content-center"
                            >
                                <a
                                    type="submit"
                                    class="btn btn-outline-success"
                                    role="button"
                                    @click="load"
                                    href="showcarg"
                                >
                                    Cargar/Editar
                                </a>
                            </div>
                        </form>
                    </div>
                </div>
            </div>
        </body>
    </html>
</template>

<script>
import axios from "axios";
export default {
    data() {
        return {
            cargo: "",
            desc: "",
        };
    },

    mounted() {
        this.load();
    },

    methods: {
        load() {
            let json = {
                cargo: this.cargo,
                desc: this.desc,
            };

            axios.post("api/cargos/createcargos", json)
            .then((data) => {
                if (data.status == 200)
                    alert("Guardado correctamente")              
            });
        },
    },
};
</script>

<style></style>


<template>
    <!DOCTYPE html>
    <html lang="es">
        <head>
            <title>Mostrar Cargos</title>
        </head>
        <body>
            <div id="table" class="container align-center col-8">
                <div class="alert alert-danger" role="alert" v-if="message">
                    {{ message }}
                </div>
                <table class="table table-bordered">
                    <thead>
                        <tr class="table-success">
                            <th scope="col">id</th>
                            <th scope="col">Cargo</th>
                            <th scope="col">Descripci√≥n</th>
                            <th scope="col">Accion</th>
                        </tr>
                    </thead>
                    <tbody>
                        <tr v-for="cargo in cargos" :key="cargo.id">
                            <td>{{ cargo.id }}</td>
                            <td>{{ cargo.cargo }}</td>
                            <td>{{ cargo.descripcion }}</td>
                            <td>
                                <a
                                    type="button"
                                    class="btn btn-warning fas fa-pencil"
                                    @click="edit(cargo.id)"
                                    to
                                >
                                </a
                                >&nbsp;
                                <a
                                    type="submit"
                                    class="btn btn-danger fas fa-trash"
                                    @click="eliminar(cargo.id)"
                                >
                                </a>
                            </td>
                        </tr>
                    </tbody>
                </table>
            </div>
        </body>
    </html>
</template>

<script>
import axios from "axios";
import { Router } from "express";

export default {
    data() {
        return {
            cargos: [],
            message: "",
        };
    },
    methods: {
        async show(id) {
            await axios
                .get("/api/cargos/index/" + id)
                .then((response) => (this.cargos = response.data));
        },
        async eliminar(id) {
            if (id) {
                await axios
                    .delete("/api/cargos/borrar/" + id)
                    .then((response) => {
                    this.message = response.data.message;
                    console.warn(response);
                    this.show();
                })
                    .catch((error) => error);
            }
        },
        async edit(id) {
            await axios
                .get("api/cargos/select/:id")
                .then((response) => console.log(response.data));
        },
    },
    mounted() {
        this.show();
    },
    components: { Router }
}

</script>
